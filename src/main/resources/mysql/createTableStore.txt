CREATE TABLE IF NOT EXISTS `mydb`.`goods`  (id INT(6)  AUTO_INCREMENT,name VARCHAR(30) NOT NULL,articul INT(30) NOT NULL,PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS `mydb`.`clients` (id INT(6)  AUTO_INCREMENT,name VARCHAR(30) NOT NULL,phoneNum VARCHAR(12) NOT NULL,PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS `mydb`.`orders` (id INT(6)  AUTO_INCREMENT,code VARCHAR(30) NOT NULL,client_id INT(30) NOT NULL,PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS `mydb`.`orders_goods` (id INT(6)  AUTO_INCREMENT, order_id INT(30) NOT NULL,goods_id INT(30) NOT NULL,PRIMARY KEY (id));
