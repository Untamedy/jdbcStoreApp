create table IF NOT EXISTS mydb.goods (id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY NOT NULL,name VARCHAR(30) NOT NULL,articul INT NOT null)
create table IF NOT EXISTS mydb.clients (id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY NOT NULL,name VARCHAR(30) NOT NULL,phoneNum VARCHAR(12) NOT NULL)
create table IF NOT EXISTS mydb.orders (id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY NOT NULL,code VARCHAR(30) NOT NULL,client_id INT NOT NULL)
create table IF NOT EXISTS mydb.orders_goods (id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY NOT NULL,order_id INT NOT NULL,goods_id INT NOT NULL)
